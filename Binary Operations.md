
True / False

NOT = !
NOT 1 = 0
NOT 0 = 1

AND = &
1 AND 1 = 1
1 AND 0 = 0
0 AND 0 = 0

OR = |
1 OR 1 = 1
1 OR 0 = 1
0 OR 0 = 0

XOR = ^ (NOT BOTH)
1 XOR 1 = 0
1 XOR 0 = 1
0 XOR 0 = 0

NAND / NOR ( NOT AND , NOT OR)
1 NAND 1 = 0
1 NAND 0 = 1
0 NAND 0 = 1
1 NOR 1    = 1
1 NOR 0    = 0
0 NOR 0    = 1
